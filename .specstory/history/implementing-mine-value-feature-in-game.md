+             var cellObject = m_GridManager.GetCellObject(kvp.Key);
+             if (cellObject != null)
+             {
+                 var cellView = cellObject.GetComponent<CellView>();
+                 if (cellView != null)
+                 {
+                     cellView.SetValue(kvp.Value);
+                 }
+             }
+         }
+     }
+     #endregion
